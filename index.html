Artisan

// Súgó
php artisan --help VAGY -h
// Kimenet letiltása / csendes mód
php artisan --quiet VAGY -q
// Laravel verzió lekérdezése
php artisan --version OR -V
// -v|vv|vvv Növeli a beszédesség mértékét: v normál mód, vv több információ, vvv debug mód
php artisan --verbose


// Lefordított class törlése
php artisan clear-compiled
// Alkalmazás ideiglenes felfüggesztése
php artisan down
// Aktuális futtatókörnyezet lekérdezése
php artisan env
// Súgó
php artisan help
// Adatbázis migráció
php artisan migrate
// Sebesség optimalizáció
php artisan optimize
// Kiszolgálása a php beépített webszerverén keresztül
php artisan serve
// Interaktív konzol
php artisan tinker
// Ideiglenes felfüggesztés vége
php artisan up


// Route -ok gyorsítótárazása
php artisan route:cache
// Route gyorsítótár törlése
php artisan route:clear
// Route -ok listázása
php artisan route


// Assetek publisholása a külső csomagokból
php artisan vendor:publish


// Lefordított view cache törlése
php artisan view:clear

Composer

// Új Laravel telepítése
composer create-project --prefer-dist laravel/laravel mappa
// Composer csomagok telepítése
composer install
// Csomagok frissítése
composer update
// Az autoloader újragenerálása
composer dump-autoload [--optimize]
// Composer verzió frissítése
composer self-update

Config

// Config érték lekérdezése (app.php -ból)
Config::get('app.locale')
// Config érték futásidejű megváltoztatása
Config::set('database.default', 'mysql');
